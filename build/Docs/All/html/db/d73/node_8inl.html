bool TOML_CALLCONV node_deep_equality(const node* lhs, const node* rhs) noexcept
{
if (lhs == rhs)
return true;
if ((!lhs != !rhs) || lhs->type() != rhs->type())
return false;
return lhs->visit(
[=](auto& l) noexcept
{
using concrete_type = remove_cvref<decltype(l)>;
return l == *(rhs->as<concrete_type>());
});
}
}
